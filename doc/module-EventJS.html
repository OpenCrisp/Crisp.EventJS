Example
var myObject = {};
var pickerCache = {};

Crisp.defineEvent( myObject );

myObject.eventListener({
  listen: function( e ) {
    console.log('action:', e.action );
    console.log('list:', JSON.stringify( e._note ) );
  }
});

var picker = myObject.eventPicker({
  cache: pickerCache
});

picker.Note({
  action: 'update'
});

picker.Talk();

// logs:
// action: 'task'
// list: '{"list":{"own":[{"action":"update"}]}}'
